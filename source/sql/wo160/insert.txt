INSERT INTO tb_mes_wo160 (
COMPANY, WLOT_NO, WO_SEQ, ITEM_CD, ITEM_QTY, STOCK_CD, BARCODE)
WITH s000 AS (SELECT * FROM tb_mes_st000 WHERE BARCODE='{0}' limit 1)
SELECT 
1000 as company,
(SELECT WLOT_NO FROM s000) as wlot_no,
IFNULL(MAX(w160.WO_SEQ)+1, 1) as wo_seq,
(SELECT ITEM_CD FROM s000) as item_cd,
(SELECT STOCK_QTY FROM s000) as item_qty,
(SELECT STOCK_CD FROM s000) as stock_cd,
'{0}' as barcode
FROM tb_mes_wo160 as w160 JOIN s000 ON w160.WLOT_NO=s000.WLOT_NO;
