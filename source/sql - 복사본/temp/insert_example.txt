INSERT INTO tb_mes_wo120 (BAD_DT, BAD_SEQ, COMPANY, BAD_CD, 
BAD_HOUR, BAD_MINUTE, BAD_SECOND, BAD_QTY, WLOT_NO)
SELECT t2.*
FROM 
(
  SELECT 1
  FROM dual
  WHERE NOT EXISTS (
    SELECT 1 FROM tb_mes_wo120 
    WHERE WLOT_NO='{0}' AND BAD_CD='DP016'
  )
) as t1,
(
  SELECT CURDATE(), IFNULL(MAX(t.BAD_SEQ)+1, 1), 1000, 'DP016', 
  LPAD(HOUR(NOW()),2,'0'), LPAD(MINUTE(NOW()),2,'0'), LPAD(SECOND(NOW()),2,'0'), 0, '{0}'
  FROM tb_mes_wo120 as t
  WHERE t.BAD_DT = CURDATE()
) as t2;
